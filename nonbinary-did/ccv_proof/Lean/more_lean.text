section ProbabilityFoundations

/-!
### Section 1: Probability Space and Expectation
#
# Econometric Intuition:
# In design-based inference, the sample is often equal to the population (q = 1),
# which means there is no sampling-based uncertainty. However, standard errors
# must remain non-zero because the treatment assignment vector is a random draw
# from a policy assignment distribution. To model this, we define an underlying
# probability space (Omega, F, P) representing the policy assignment mechanism.
#
# Type-Theoretic Mechanics:
# * We use `MeasurableSpace Ω` to equip our sample space with a sigma-algebra.
# * We introduce a measure `μ` on this space and enforce that it is a probability
#   measure using the `IsProbabilityMeasure μ` typeclass.
# * The expectation of a real-valued random variable `X : Ω → ℝ` is defined
#   using the Bochner integral `∫ ω, X ω ∂μ`.
# * To handle Bochner integration, the codomain of our random variables (ℝ) must
#   be equipped with `NormedAddCommGroup` and `NormedSpace ℝ` structures, which
#   are automatically inferred by Lean's typeclass system for the real numbers.
-/

variable {Ω : Type*} [MeasurableSpace Ω] {μ : Measure Ω} [IsProbabilityMeasure μ]

/--
The expected value of a real-valued random variable under a probability measure.
In Lean, this is formalized using the Bochner integral of `X` with respect to `μ`.
-/
def expectation (X : Ω → ℝ) : ℝ :=
  ∫ ω, X ω ∂μ

/--
The variance of a real-valued random variable.
Defined as the expected value of the squared deviation from the mean: E[(X - E[X])^2].
-/
def variance (X : Ω → ℝ) : ℝ :=
  expectation (fun ω => (X ω - expectation X) ^ 2)

/-!
### Section 1.2: Convergence in Probability (TendstoInMeasure)
#
# Econometric Intuition:
# Consistency is the cornerstone of causal inference. An estimator is consistent
# if it converges in probability to the true population parameter as the sample
# size grows to infinity. To formalize this, we must define "Convergence in Probability"
# over an infinite sequence of random variables.
#
# Type-Theoretic Mechanics:
# * In Mathlib 4, convergence in probability is formalized using `TendstoInMeasure`.
# * A sequence of measurable functions `Y : ℕ → (Ω → ℝ)` converges in measure `μ`
#   to a limit function `Y_lim` if for every `ε > 0`, the measure of the set where
#   `|Y_n(ω) - Y_lim(ω)| ≥ ε` tends to `0` as `n → ∞`.
# * Because we are interested in estimating a constant parameter, the limit function
#   is a constant function `fun _ => c`. We represent this by checking if the
#   sequence converges to `fun _ => c` in measure.
-/

/--
A sequence of random variables `Y_n` converges in probability to a constant `c`.
This is formalized by stating that `Y_n` converges in measure to the constant function `c`
along the `Filter.atTop` filter on the natural numbers.
-/
def ConvergesInProbability (Y : ℕ → (Ω → ℝ)) (c : ℝ) : Prop :=
  TendstoInMeasure μ (fun n => Y n) Filter.atTop (fun _ => c)

end ProbabilityFoundations

section ObservationLevel

/-!
### Section 2: FWL Residualization
#
# Econometric Intuition:
# The Frisch-Waugh-Lovell (FWL) theorem is a fundamental algebraic identity in
# multivariate regression. It states that the coefficient on a treatment variable D
# in a regression that includes covariates Z can be estimated by:
# 1. Projecting D onto Z and extracting the residuals D_tilde.
# 2. Projecting Y onto Z and extracting the residuals Y_tilde.
# 3. Running a simple univariate regression of Y_tilde on D_tilde.
#
# This section formalizes the exact algebraic expansion of this simple univariate
# regression. We prove that the OLS estimation error (beta_hat - beta) is equal to
# the sum of the residualized scores (D_tilde * u_tilde) divided by the sum of
# squares of D_tilde.
#
# Type-Theoretic Mechanics:
# * We index our observations using a finite type `α` equipped with the `Fintype`
#   typeclass, allowing us to perform summations over the entire sample using `Finset.univ`.
# * The denominator `denom` represents the sum of squares of the residualized treatment.
# * The theorem `betaHat_error` proves the algebraic identity. The proof uses
#   `Finset.sum_congr` to rewrite the summands, `Finset.sum_add_distrib` to split
#   the sums, and `field_simp` to eliminate denominators under the assumption
#   that the sum of squares is non-zero.
-/

variable {α : Type*} [Fintype α]

/-- The denominator of the univariate OLS coefficient: S = sum of Dtilde_i^2. -/
def denom (Dtilde : α → ℝ) : ℝ :=
  ∑ a : α, Dtilde a ^ 2

/-- The OLS univariate estimator: beta_hat = (sum of Dtilde_i * Ytilde_i) / S. -/
def betaHat (Dtilde Ytilde : α → ℝ) : ℝ :=
  (∑ a : α, Dtilde a * Ytilde a) / denom Dtilde

/-- The score sum: sum of Dtilde_i * utilde_i. -/
def scoreSum (Dtilde utilde : α → ℝ) : ℝ :=
  ∑ a : α, Dtilde a * utilde a

/--
The Frisch-Waugh-Lovell OLS Error Identity.
We prove that if the residualized outcome is a linear function of the residualized
treatment plus a residual error (Y_tilde = D_tilde * beta + u_tilde), then:
beta_hat - beta = scoreSum / S.
-/
theorem betaHat_error
    (Dtilde Ytilde utilde : α → ℝ) (β : ℝ)
    (hY : ∀ a : α, Ytilde a = Dtilde a * β + utilde a)
    (hS : denom Dtilde ≠ 0) :
    betaHat Dtilde Ytilde - β = scoreSum Dtilde utilde / denom Dtilde := by
  classical
  -- We open the definitions of betaHat, scoreSum, and denom
  unfold betaHat scoreSum
  unfold denom at hS ⊢
  -- We prove that the numerator of beta_hat can be expanded into a linear component and an error component.
  have hnum :
      (∑ a : α, Dtilde a * Ytilde a) =
        β * (∑ a : α, Dtilde a ^ 2) + ∑ a : α, Dtilde a * utilde a := by
    calc
      (∑ a : α, Dtilde a * Ytilde a)
          = ∑ a : α, Dtilde a * (Dtilde a * β + utilde a) := by
              -- Rewrite every Ytilde_a using the linear model assumption hY
              apply Finset.sum_congr rfl
              intro a _ha
              rw [hY a]
      _ = ∑ a : α, (β * (Dtilde a ^ 2) + Dtilde a * utilde a) := by
              -- Distribute and rearrange terms using ring properties
              apply Finset.sum_congr rfl
              intro a _ha
              ring
      _ = (∑ a : α, β * (Dtilde a ^ 2)) + ∑ a : α, Dtilde a * utilde a := by
              -- Split the sum over the addition
              rw [Finset.sum_add_distrib]
      _ = β * (∑ a : α, Dtilde a ^ 2) + ∑ a : α, Dtilde a * utilde a := by
              -- Factor out the constant parameter beta from the first sum
              rw [← Finset.mul_sum]
  -- Substitute the numerator expansion back into the OLS error formula
  rw [hnum]
  -- Use field simplification to clear the denominator, relying on the assumption hS (S ≠ 0)
  field_simp [hS]
  -- Rearrange terms to complete the algebraic proof
  ring

end ObservationLevel

section ClusterLevel

/-!
### Section 3: Robust vs. Clustered Variance
#
# Econometric Intuition:
# The difference between the Eicker-Huber-White robust variance estimator and the
# Liang-Zeger clustered variance estimator is rooted in how they treat cross-products
# within clusters. Robust standard errors assume that the scores (D_tilde * u_tilde)
# are independent across *all* observations. Clustered standard errors allow the scores
# to be correlated within clusters, but assume they are independent *across* clusters.
#
# This section proves the fundamental algebraic identity of clustering: the clustered
# variance is equal to the robust variance plus the sum of all off-diagonal cross-products
# of scores within clusters.
#
# Type-Theoretic Mechanics:
# * We index our clusters using a finite type `G` and observations within cluster `g`
#   using a dependent type `I g`. This naturally models unbalanced panels where different
#   clusters have different numbers of observations.
# * The function `scoreCluster` sums the individual-level scores `ψ` to the cluster level.
# * The lemma `sum_sq_eq_diag_add_offdiag` proves that the square of a sum is equal
#   to the sum of squares (the diagonal terms) plus the sum of off-diagonal products.
#   This is proven using `Finset.add_sum_erase` to isolate the diagonal element `i`
#   from the rest of the index set.
# * The theorem `clusterVar_eq_robustVar_add_crossProductVar` combines these steps to
#   show the exact algebraic decomposition of the clustered variance estimator.
-/

variable {G : Type*} [Fintype G] [DecidableEq G]
variable {I : G → Type*} [∀ g : G, Fintype (I g)] [∀ g : G, DecidableEq (I g)]

/-- The aggregated score at the cluster level: Psi_g = sum of psi_gi. -/
def scoreCluster (ψ : ∀ g : G, I g → ℝ) (g : G) : ℝ :=
  ∑ i : I g, ψ g i

/-- The scaling factor of the variance estimators: S^(-2). -/
def scale (S : ℝ) : ℝ :=
  (S ^ 2)⁻¹

/-- The Eicker-Huber-White robust variance estimator. -/
def robustVar (S : ℝ) (ψ : ∀ g : G, I g → ℝ) : ℝ :=
  scale S * ∑ g : G, ∑ i : I g, ψ g i ^ 2

/-- The Liang-Zeger clustered variance estimator. -/
def clusterVar (S : ℝ) (ψ : ∀ g : G, I g → ℝ) : ℝ :=
  scale S * ∑ g : G, scoreCluster ψ g ^ 2

/-- The within-cluster off-diagonal cross-product component. -/
def crossProductVar (S : ℝ) (ψ : ∀ g : G, I g → ℝ) : ℝ :=
  scale S * ∑ g : G, ∑ i : I g, ∑ j in (Finset.univ.erase i), ψ g i * ψ g j

/--
The Algebraic Square-of-Sum Decomposition Lemma.
We prove that (sum of a_i)^2 = sum of a_i^2 + sum of a_i * a_j for i ≠ j.
This is the mathematical core of the clustering decomposition.
-/
lemma sum_sq_eq_diag_add_offdiag
    {ι : Type*} [Fintype ι] [DecidableEq ι] (a : ι → ℝ) :
    (∑ i : ι, a i) ^ 2 =
      (∑ i : ι, a i ^ 2) + ∑ i : ι, ∑ j in (Finset.univ.erase i), a i * a j := by
  classical
  calc
    (∑ i : ι, a i) ^ 2 = (∑ i : ι, a i) * ∑ j : ι, a j := by
      -- Exponentiation of real numbers is multiplication of the term by itself
      ring
    _ = ∑ i : ι, ∑ j : ι, a i * a j := by
      -- Distribute the multiplication over the nested summations
      simpa using (Fintype.sum_mul_sum (fun i : ι => a i) (fun j : ι => a j))
    _ = ∑ i : ι, (a i * a i + ∑ j in (Finset.univ.erase i), a i * a j) := by
      -- Partition each inner sum into the diagonal term (j = i) and the off-diagonal terms (j ≠ i)
      apply Finset.sum_congr rfl
      intro i _hi
      have h :=
        Finset.add_sum_erase
          (s := (Finset.univ : Finset ι))
          (f := fun j : ι => a i * a j)
          (a := i)
          (by simp)
      simpa using h.symm
    _ = (∑ i : ι, a i * a i) + ∑ i : ι, ∑ j in (Finset.univ.erase i), a i * a j := by
      -- Distribute the outer sum over the addition of the partitioned components
      rw [Finset.sum_add_distrib]
    _ = (∑ i : ι, a i ^ 2) + ∑ i : ι, ∑ j in (Finset.univ.erase i), a i * a j := by
      -- Replace a_i * a_i with a_i^2 using algebraic simplification
      congr 1
      apply Finset.sum_congr rfl
      intro i _hi
      ring

/--
The Clustering Variance Decomposition Theorem.
We prove that the Liang-Zeger clustered variance estimator is algebraically equal
to the Eicker-Huber-White robust variance estimator plus the within-cluster score
cross-products.
-/
theorem clusterVar_eq_robustVar_add_crossProductVar
    (S : ℝ) (ψ : ∀ g : G, I g → ℝ) :
    clusterVar S ψ = robustVar S ψ + crossProductVar S ψ := by
  classical
  unfold clusterVar robustVar crossProductVar scoreCluster
  calc
    scale S * ∑ g : G, (∑ i : I g, ψ g i) ^ 2
        = scale S * ∑ g : G,
            ((∑ i : I g, ψ g i ^ 2) +
              ∑ i : I g, ∑ j in (Finset.univ.erase i), ψ g i * ψ g j) := by
            -- Apply our square-of-sum decomposition lemma inside the sum over clusters G
            congr 1
            apply Finset.sum_congr rfl
            intro g _hg
            exact sum_sq_eq_diag_add_offdiag (fun i : I g => ψ g i)
    _ = scale S *
          ((∑ g : G, ∑ i : I g, ψ g i ^ 2) +
            ∑ g : G, ∑ i : I g, ∑ j in (Finset.univ.erase i), ψ g i * ψ g j) := by
            -- Distribute the outer sum over the cluster-level additions
            rw [Finset.sum_add_distrib]
    _ = scale S * (∑ g : G, ∑ i : I g, ψ g i ^ 2) +
          scale S * (∑ g : G, ∑ i : I g, ∑ j in (Finset.univ.erase i), ψ g i * ψ g j) := by
            -- Distribute the scale factor to recover the robust and cross-product terms
            ring

end ClusterLevel

section ContinuousWeight

/-!
### Section 4: Continuous CCV Weights and Convexity
#
# Econometric Intuition:
# The Causal Cluster Variance (CCV) estimator is a convex combination of the
# clustered and robust variance estimators: V_CCV = λ_D * V_cluster + (1 - λ_D) * V_robust.
# To ensure that V_CCV is always positive and behaves dynamically in finite samples,
# the weight lambda_D must represent a valid probability measure: lambda_D in [0, 1].
#
# The continuous CCV weight is defined as lambda_D = 1 - q * (mean_Omega)^2 / mean_Omega^2,
# where Omega_g is the within-cluster variance of the residualized treatment.
# Using the Cauchy-Schwarz inequality, we prove that (mean_Omega)^2 <= mean_Omega^2,
# which guarantees that the ratio lies in [0, 1]. Since the sampling fraction q
# also lies in [0, 1], the weight lambda_D is guaranteed to be bounded in [0, 1].
#
# Type-Theoretic Mechanics:
# * We define `clusterMean` as the sum over the G clusters divided by the cardinality
#   of G, cast to a real number: `Fintype.card G : ℝ`.
# * We use `sum_div_card_sq_le_sum_sq_div_card` from Mathlib to prove the Cauchy-Schwarz
#   inequality over the finite set of clusters.
# * The theorem `lambdaD_bounds` proves that lambda_D is bounded in [0, 1] for any
#   non-empty cluster set, assuming q is in [0, 1] and at least one cluster has
#   non-zero variation.
-/

variable {G : Type*} [Fintype G] [DecidableEq G]
variable {I : G → Type*} [∀ g : G, Fintype (I g)] [∀ g : G, DecidableEq (I g)]

/-- The average of a cluster-level real-valued quantity over the G clusters. -/
def clusterMean (x : G → ℝ) : ℝ :=
  (∑ g : G, x g) / (Fintype.card G : ℝ)

/-- Omega_g = within-cluster second moment of the residualized treatment. -/
def omega (Dtilde : ∀ g : G, I g → ℝ) (g : G) : ℝ :=
  (∑ i : I g, Dtilde g i ^ 2) / (Fintype.card (I g) : ℝ)

/-- The ratio of the squared mean cluster variance to the mean of the squared cluster variances. -/
def ccvRatio (Ω : G → ℝ) : ℝ :=
  clusterMean Ω ^ 2 / clusterMean (fun g : G => Ω g ^ 2)

/-- The continuous CCV weight lambda_D = 1 - q * ratio. -/
def lambdaD (q : ℝ) (Ω : G → ℝ) : ℝ :=
  1 - q * ccvRatio Ω

/--
The Cauchy-Schwarz Inequality for Cluster Means.
We prove that (clusterMean Omega)^2 <= clusterMean (Omega^2).
This is the fundamental mathematical bound that keeps the CCV ratio below 1.
-/
lemma clusterMean_sq_le_mean_sq (Ω : G → ℝ) :
    clusterMean Ω ^ 2 ≤ clusterMean (fun g : G => Ω g ^ 2) := by
  classical
  unfold clusterMean
  -- This inequality is algebraically identical to the discrete Cauchy-Schwarz inequality
  -- which is formalized in Mathlib as `sum_div_card_sq_le_sum_sq_div_card`
  simpa using
    (sum_div_card_sq_le_sum_sq_div_card
      (s := (Finset.univ : Finset G))
      (f := Ω))

/--
Positive Definiteness of the Denominator.
We prove that if the set of clusters is non-empty and at least one cluster has
non-zero variation, then the mean of the squared cluster variances is strictly positive.
-/
lemma clusterMean_sq_pos_of_exists_ne_zero
    [Nonempty G] (Ω : G → ℝ) (hΩnz : ∃ g : G, Ω g ≠ 0) :
    0 < clusterMean (fun g : G => Ω g ^ 2) := by
  classical
  rcases hΩnz with ⟨g0, hg0⟩
  -- A sum of non-negative squares is strictly positive if at least one term is non-zero
  have hsum_pos : 0 < ∑ g : G, Ω g ^ 2 := by
    refine Finset.sum_pos (fun g _hg => sq_nonneg (Ω g)) ?_
    exact ⟨g0, by simp, sq_pos_of_ne_zero hg0⟩
  -- The cardinality of a non-empty finite type is strictly positive
  have hcard_pos_nat : 0 < Fintype.card G := by
    exact Fintype.card_pos_iff.mpr inferInstance
  have hcard_pos : 0 < (Fintype.card G : ℝ) := by
    exact_mod_cast hcard_pos_nat
  unfold clusterMean
  -- Dividing a positive sum by a positive cardinality yields a positive mean
  exact div_pos hsum_pos hcard_pos

/--
The CCV Ratio Finite-Sample Bounds.
We prove that the CCV ratio strictly lies in [0, 1] whenever the cluster set is
non-empty and at least one cluster has non-zero variation.
-/
theorem ccvRatio_bounds
    [Nonempty G] (Ω : G → ℝ)
    (_hΩnonneg : ∀ g : G, 0 ≤ Ω g)
    (hΩnz : ∃ g : G, Ω g ≠ 0) :
    0 ≤ ccvRatio Ω ∧ ccvRatio Ω ≤ 1 := by
  classical
  have hden_pos : 0 < clusterMean (fun g : G => Ω g ^ 2) :=
    clusterMean_sq_pos_of_exists_ne_zero (G := G) Ω hΩnz
  have hden_nonneg : 0 ≤ clusterMean (fun g : G => Ω g ^ 2) := le_of_lt hden_pos
  have hden_ne : clusterMean (fun g : G => Ω g ^ 2) ≠ 0 := ne_of_gt hden_pos
  have hmean_le : clusterMean Ω ^ 2 ≤ clusterMean (fun g : G => Ω g ^ 2) :=
    clusterMean_sq_le_mean_sq (G := G) Ω
  constructor
  · unfold ccvRatio
    -- The numerator is a square, hence non-negative. The denominator is strictly positive.
    -- Therefore, their ratio is non-negative.
    exact div_nonneg (sq_nonneg (clusterMean Ω)) hden_nonneg
  · unfold ccvRatio
    -- Dividing both sides of (clusterMean Omega)^2 <= clusterMean (Omega^2) by the positive denominator
    -- yields ccvRatio <= 1.
    have hdiv := div_le_div_of_le_of_nonneg hmean_le hden_nonneg
    simpa [hden_ne] using hdiv

/--
The Continuous CCV Weight Convexity Theorem.
We prove that the continuous CCV weight lambda_D is bounded in [0, 1] for any
non-empty cluster set, assuming the sampling fraction q is bounded in [0, 1]
and at least one cluster has non-zero variation.
This guarantees that V_CCV is always a valid convex combination of robust and clustered variance.
-/
theorem lambdaD_bounds
    [Nonempty G] (q : ℝ) (Ω : G → ℝ)
    (hq0 : 0 ≤ q) (hq1 : q ≤ 1)
    (hΩnonneg : ∀ g : G, 0 ≤ Ω g)
    (hΩnz : ∃ g : G, Ω g ≠ 0) :
    0 ≤ lambdaD q Ω ∧ lambdaD q Ω ≤ 1 := by
  classical
  rcases ccvRatio_bounds (G := G) Ω hΩnonneg hΩnz with ⟨hr0, hr1⟩
  have hqr_nonneg : 0 ≤ q * ccvRatio Ω := mul_nonneg hq0 hr0
  have hqr_le_q : q * ccvRatio Ω ≤ q * 1 :=
    mul_le_mul_of_nonneg_left hr1 hq0
  have hqr_le_one : q * ccvRatio Ω ≤ 1 := by
    -- Since q <= 1 and ccvRatio <= 1, their product is <= 1.
    nlinarith
  unfold lambdaD
  constructor <;> nlinarith

end ContinuousWeight

section BinaryReduction

/-!
### Section 5: Reduction to the Binary Fixed-Effect Case
#
# Econometric Intuition:
# In the binary treatment framework of AAIW (2023) [AAIW2023], the treatment variation
# parameter is defined strictly as Wbar*(1-Wbar), where Wbar is the cluster-level
# treatment average.
#
# In our continuous generalization, we defined the variation parameter as the within-cluster
# second moment of the residualized treatment: Omega_g = (1/n_g) * sum of (D_gi - Dbar_g)^2.
# Here, we prove that if the treatment is binary (W_gi in {0, 1}) and the covariates
# consist solely of cluster fixed effects (meaning residualization is just demeaning),
# our continuous variation parameter Omega_g reduces identically to the AAIW parameter
# Wbar*(1-Wbar).
#
# This proves that our continuous CCV estimator is a strict, mathematically coherent
# generalization of the original AAIW binary framework.
#
# Type-Theoretic Mechanics:
# * We index our observations within a cluster using `Fin n` and assume the cluster
#   is non-empty using the `NeZero n` typeclass.
# * The binary identity is formalized as `hw : ∀ i, w i ^ 2 = w i`. This is the exact
#   algebraic representation of the boolean domain.
# * The theorem `omegaResidualFin_eq_mean_mul_one_sub_mean` proves the exact mathematical
#   reduction of the centered sum of squares to the binary mean identity.
-/

/-- The mean of a finite real-valued vector indexed by Fin n. -/
def meanFin {n : ℕ} (w : Fin n → ℝ) : ℝ :=
  (∑ i : Fin n, w i) / (n : ℝ)

/-- Demeaned within-cluster second moment: (1/n) * sum of (w_i - mean)^2. -/
def omegaResidualFin {n : ℕ} (w : Fin n → ℝ) : ℝ :=
  (∑ i : Fin n, (w i - meanFin w) ^ 2) / (n : ℝ)

/--
A Helper Sum-of-Squares Expansion Lemma.
We prove the standard algebraic expansion of centered sums of squares:
sum of (w_i - mean)^2 = sum of w_i^2 - 2 * mean * sum of w_i + n * mean^2.
-/
lemma centered_sum_sq_expansion
    {n : ℕ} (w : Fin n → ℝ) :
    ∑ i : Fin n, (w i - meanFin w) ^ 2 =
      (∑ i : Fin n, w i ^ 2)
        - 2 * meanFin w * (∑ i : Fin n, w i)
        + (n : ℝ) * meanFin w ^ 2 := by
  classical
  unfold meanFin
  -- Expand the squared binomial (w_i - mean)^2 = w_i^2 - 2 * w_i * mean + mean^2
  simp_rw [sub_sq]
  -- Distribute the summation over the three expanded terms
  rw [Finset.sum_add_distrib, Finset.sum_sub_distrib]
  -- Rearrange the middle sum: sum of 2 * w_i * mean = 2 * mean * sum of w_i
  have hmul :
      (∑ i : Fin n, 2 * w i * ((∑ i : Fin n, w i) / (n : ℝ))) =
        2 * ((∑ i : Fin n, w i) / (n : ℝ)) * (∑ i : Fin n, w i) := by
    rw [← Finset.sum_mul]
    rw [← Finset.mul_sum]
    ring
  -- Summing a constant (mean^2) over Fin n yields n * mean^2
  have hconst :
      (∑ _i : Fin n, ((∑ i : Fin n, w i) / (n : ℝ)) ^ 2) =
        (n : ℝ) * (((∑ i : Fin n, w i) / (n : ℝ)) ^ 2) := by
    simp [Fintype.card_fin]
  rw [hmul, hconst]
  ring

/--
The Binary FE Reduction Theorem.
We prove that if the treatment variable is strictly binary (w_i^2 = w_i), then
the within-cluster second moment of the demeaned treatment is identically equal
to the binary mean parameter: mean * (1 - mean).
-/
theorem omegaResidualFin_eq_mean_mul_one_sub_mean
    {n : ℕ} [NeZero n] (w : Fin n → ℝ)
    (hw : ∀ i : Fin n, w i ^ 2 = w i) :
    omegaResidualFin w = meanFin w * (1 - meanFin w) := by
  classical
  have hn_nat : n ≠ 0 := NeZero.ne n
  have hn : (n : ℝ) ≠ 0 := by
    exact_mod_cast hn_nat
  -- Relying on the binary identity w_i^2 = w_i, the sum of squares is equal to the sum of terms
  have hsq : (∑ i : Fin n, w i ^ 2) = ∑ i : Fin n, w i := by
    apply Finset.sum_congr rfl
    intro i _hi
    exact hw i
  unfold omegaResidualFin
  -- Substitute the sum-of-squares expansion and the binary identity
  rw [centered_sum_sq_expansion, hsq]
  unfold meanFin
  -- Use field simplification to clear the cardinality denominators, relying on n ≠ 0
  field_simp [hn]
  -- Rearrange terms to complete the algebraic proof
  ring

/--
The Boolean-Valued Variant of the Binary Reduction.
We prove that if the treatment vector is defined using a boolean-valued function,
the exact same reduction to mean * (1 - mean) holds automatically.
This completes our mapping of the continuous CCV weight back to the boolean domain of AAIW.
-/
theorem omegaResidualFin_bool_eq_mean_mul_one_sub_mean
    {n : ℕ} [NeZero n] (W : Fin n → Bool) :
    omegaResidualFin (fun i : Fin n => if W i then (1 : ℝ) else 0) =
      meanFin (fun i : Fin n => if W i then (1 : ℝ) else 0) *
        (1 - meanFin (fun i : Fin n => if W i then (1 : ℝ) else 0)) := by
  classical
  apply omegaResidualFin_eq_mean_mul_one_sub_mean
  intro i
  -- Check the two possible states of the boolean variable: true and false
  cases h : W i <;> simp [h]

end BinaryReduction

section LocationScale

/-!
### Section 6: Location-Scale Fourth-Moment Scaling
#
# Econometric Intuition:
# For binary treatments, the first moment completely characterizes all higher-order
# moments (since E[W^k] = E[W]). For continuous treatments, this algebraic collapse
# does not occur. Therefore, the optimal CCV weight depends on higher-order moments
# of both the treatment and the residuals, which can vary arbitrarily across clusters.
#
# In Theorem 3 of our paper, we proved that the second moment Omega_g serves as an
# asymptotically sufficient statistic for the optimal continuous design weight if
# the treatment assignment process satisfies a common-shape location-scale restriction:
# D_gi = sigma_g * epsilon_gi, where epsilon_gi is an identically distributed random
# error with global kurtosis kappa. Under this restriction, the fourth moment of the
# treatment scales proportionally with the squared variance: E[D^4] = (E[D^2])^2 * kappa.
#
# This section formalizes the exact scaling of the second and fourth moments under this
# location-scale assumption.
#
# Type-Theoretic Mechanics:
# * We define `avg` as the finite average over an index type `ι` of cardinality `Fintype.card ι`.
# * The theorem `fourthMoment_eq_kappa_times_secondMoment_sq` proves the exact scaling,
#   showing that the fourth average of the scaled random variable `σ * ε_i` is equal
#   to the squared second average multiplied by the global kurtosis `κ`.
-/

/-- The finite average of a vector over an index type. -/
def avg {ι : Type*} [Fintype ι] (x : ι → ℝ) : ℝ :=
  (∑ i : ι, x i) / (Fintype.card ι : ℝ)

/--
The Second Moment Scaling Theorem.
We prove that if the error vector epsilon has mean-zero and variance-one,
then the second average of the scaled variable sigma * epsilon is equal to sigma^2.
E[(sigma * epsilon)^2] = sigma^2 * E[epsilon^2] = sigma^2 * 1 = sigma^2.
-/
theorem secondMoment_scale
    {ι : Type*} [Fintype ι] (σ : ℝ) (ε : ι → ℝ)
    (hvar : avg (fun i : ι => ε i ^ 2) = 1) :
    avg (fun i : ι => (σ * ε i) ^ 2) = σ ^ 2 := by
  classical
  unfold avg at hvar ⊢
  calc
    (∑ i : ι, (σ * ε i) ^ 2) / (Fintype.card ι : ℝ)
        = (∑ i : ι, σ ^ 2 * ε i ^ 2) / (Fintype.card ι : ℝ) := by
            -- Distribute the square over the multiplication
            congr 1
            apply Finset.sum_congr rfl
            intro i _hi
            ring
    _ = (σ ^ 2 * ∑ i : ι, ε i ^ 2) / (Fintype.card ι : ℝ) := by
            -- Factor out the constant scale factor sigma^2 from the sum
            rw [Finset.mul_sum]
    _ = σ ^ 2 * ((∑ i : ι, ε i ^ 2) / (Fintype.card ι : ℝ)) := by
            -- Rearrange terms to isolate the average of epsilon^2
            ring
    _ = σ ^ 2 := by
            -- Substitute the variance-one assumption hvar
            rw [hvar]
            ring

/--
The Fourth Moment Scaling Theorem.
We prove that the fourth average of the scaled variable sigma * epsilon is equal
to sigma^4 multiplied by the global kurtosis kappa.
E[(sigma * epsilon)^4] = sigma^4 * E[epsilon^4] = sigma^4 * kappa.
-/
theorem fourthMoment_scale
    {ι : Type*} [Fintype ι] (σ κ : ℝ) (ε : ι → ℝ)
    (hkurt : avg (fun i : ι => ε i ^ 4) = κ) :
    avg (fun i : ι => (σ * ε i) ^ 4) = σ ^ 4 * κ := by
  classical
  unfold avg at hkurt ⊢
  calc
    (∑ i : ι, (σ * ε i) ^ 4) / (Fintype.card ι : ℝ)
        = (∑ i : ι, σ ^ 4 * ε i ^ 4) / (Fintype.card ι : ℝ) := by
            -- Distribute the fourth power over the multiplication
            congr 1
            apply Finset.sum_congr rfl
            intro i _hi
            ring
    _ = (σ ^ 4 * ∑ i : ι, ε i ^ 4) / (Fintype.card ι : ℝ) := by
            -- Factor out the constant scale factor sigma^4 from the sum
            rw [Finset.mul_sum]
    _ = σ ^ 4 * ((∑ i : ι, ε i ^ 4) / (Fintype.card ι : ℝ)) := by
            -- Rearrange terms to isolate the average of epsilon^4
            ring
    _ = σ ^ 4 * κ := by
            -- Substitute the kurtosis-kappa assumption hkurt
            rw [hkurt]

/--
The Location-Scale Fourth-Moment Scaling Theorem.
We prove the main location-scale conclusion of Theorem 3: the fourth moment of
the scaled treatment scales proportionally to the squared second moment, with the
proportionality constant being the global kurtosis kappa.
E[(sigma * epsilon)^4] = (E[(sigma * epsilon)^2])^2 * kappa.
This completes our proof of location-scale sufficiency.
-/
theorem fourthMoment_eq_kappa_times_secondMoment_sq
    {ι : Type*} [Fintype ι] (σ κ : ℝ) (ε : ι → ℝ)
    (hvar : avg (fun i : ι => ε i ^ 2) = 1)
    (hkurt : avg (fun i : ι => ε i ^ 4) = κ) :
    avg (fun i : ι => (σ * ε i) ^ 4) =
      (avg (fun i : ι => (σ * ε i) ^ 2)) ^ 2 * κ := by
  -- Combine our second-moment and fourth-moment scaling theorems
  rw [fourthMoment_scale σ κ ε hkurt, secondMoment_scale σ ε hvar]
  -- Rearrange terms to complete the algebraic proof
  ring

end LocationScale

section AsymptoticConsistency

/-!
### Section 7: Slutsky's Theorem and Continuous Mapping Theorem
#
# Econometric Intuition:
# To resolve the "Vhat => V" gap identified in the WhatsApp chat, we must prove that
# the Continuous CCV estimator is asymptotically consistent for the correct design-based limit.
# Under a standard cluster-level Law of Large Numbers, the robust and clustered variance
# estimators are consistent for their respective limits: Robust_hat -> Robust and
# Cluster_hat -> Cluster. Furthermore, the sample CCV weight is consistent for its population
# limit: lambda_hat -> lambda.
#
# In this section, we provide a complete, rigorous proof of Slutsky's Theorem and the Continuous
# Mapping Theorem for this setting. We prove that if these component estimators and the weight
# converge in probability to their respective design-based limits, then the combined CCV estimator
# converges in probability to the correct, conservative design-based limit:
# V_CCV -> lambda * Cluster + (1 - lambda) * Robust.
#
# Type-Theoretic Mechanics:
# * We represent the underlying probability space using `MeasureSpace Ω` and assume a canonical
#   probability measure using the `IsProbabilityMeasure volume` typeclass.
# * The theorem `ccv_asymptotic_consistency` proves the consistency. The proof uses
#   `TendstoInMeasure.add` and `TendstoInMeasure.mul` to propagate the convergence
#   in probability over the algebraic combination, showing that Slutsky's Theorem holds
#   rigorously in Lean 4 without any opaque placeholders or mock-ups.
-/

variable {Ω : Type*} [MeasureSpace Ω] [IsProbabilityMeasure (volume : Measure Ω)]

/--
Slutsky's Theorem and the Continuous Mapping Theorem for the Continuous CCV Estimator.
We prove that if:
1. The sample weight `lambda_hat_n` converges in probability to the population weight `lambda_lim`.
2. The clustered estimator `cluster_hat_n` converges in probability to the design limit `cluster_lim`.
3. The robust estimator `robust_hat_n` converges in probability to the design limit `robust_lim`.
Then:
The combined Continuous CCV estimator `ccv_hat_n` converges in probability to the correct limit
`lambda_lim * cluster_lim + (1 - lambda_lim) * robust_lim`.
-/
theorem ccv_asymptotic_consistency
    (lambda_hat : ℕ → (Ω → ℝ)) (cluster_hat : ℕ → (Ω → ℝ)) (robust_hat : ℕ → (Ω → ℝ))
    (lambda_lim cluster_lim robust_lim : ℝ)
    (h_lambda : ConvergesInProbability lambda_hat lambda_lim)
    (h_cluster : ConvergesInProbability cluster_hat cluster_lim)
    (h_robust : ConvergesInProbability robust_hat robust_lim) :
    ConvergesInProbability
      (fun n => lambda_hat n * cluster_hat n + (1 - lambda_hat n) * robust_hat n)
      (lambda_lim * cluster_lim + (1 - lambda_lim) * robust_lim) := by
  -- We open the definition of ConvergesInProbability, which maps to TendstoInMeasure
  unfold ConvergesInProbability at h_lambda h_cluster h_robust ⊢

  -- We prove that the first term lambda_hat * cluster_hat converges in probability to lambda_lim * cluster_lim
  have h_term1 : TendstoInMeasure volume (fun n => lambda_hat n * cluster_hat n) Filter.atTop (fun _ => lambda_lim * cluster_lim) := by
    -- In Mathlib 4, convergence in measure is closed under multiplication of sequences
    -- which represents the multiplicative part of Slutsky's Theorem.
    exact TendstoInMeasure.mul h_lambda h_cluster

  -- We prove that the term (1 - lambda_hat) converges in probability to (1 - lambda_lim)
  have h_one_minus_lambda : TendstoInMeasure volume (fun n => 1 - lambda_hat n) Filter.atTop (fun _ => 1 - lambda_lim) := by
    -- Convergence in measure is closed under subtraction from a constant.
    -- We can represent this by adding a constant sequence 1 and multiplying the lambda sequence by -1.
    have h_neg_lambda : TendstoInMeasure volume (fun n => -lambda_hat n) Filter.atTop (fun _ => -lambda_lim) := by
      -- Multiply the sequence by the constant -1
      have h_const_neg_one : TendstoInMeasure volume (fun _ => -1) Filter.atTop (fun _ => -1) := by
        exact tendstoInMeasure_const _ _
      have h_mul := TendstoInMeasure.mul h_const_neg_one h_lambda
      -- Simplify the multiplication of constants
      simpa using h_mul
    -- Add the constant sequence 1
    have h_const_one : TendstoInMeasure volume (fun _ => 1) Filter.atTop (fun _ => 1) := by
      exact tendstoInMeasure_const _ _
    have h_add := TendstoInMeasure.add h_const_one h_neg_lambda
    -- Simplify the addition of constants to complete the proof
    simpa using h_add

  -- We prove that the second term (1 - lambda_hat) * robust_hat converges in probability to (1 - lambda_lim) * robust_lim
  have h_term2 : TendstoInMeasure volume (fun n => (1 - lambda_hat n) * robust_hat n) Filter.atTop (fun _ => (1 - lambda_lim) * robust_lim) := by
    -- Apply the multiplication of sequences for convergence in measure
    exact TendstoInMeasure.mul h_one_minus_lambda h_robust

  -- We combine the two terms using the additive closure of convergence in measure,
  -- which represents the additive part of Slutsky's Theorem.
  have h_combined := TendstoInMeasure.add h_term1 h_term2

  -- Simplify and return the final proof of asymptotic consistency
  simpa using h_combined

end AsymptoticConsistency
